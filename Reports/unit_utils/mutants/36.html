def bubble_sort(arr):
    '''Sorts a list using Bubble Sort.'''
    if not (isinstance(arr, list)):
        return None
    
    n = len(arr)
    for i in range(n):
        for j in range(0, (n - i) - 1):
            if arr[j] > arr[j + 1]:
                (arr[j], arr[j + 1]) = (arr[j + 1], arr[j])
    return arr
