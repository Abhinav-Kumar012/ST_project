def polygon_area(points):
    '''Calculates area of a polygon using Shoelace formula.'''
    n = len(points)
    if n < 3:
        return 0
    
    area = 0.0
    for i in range(n):
        j = (i + 1) % n
        area += points[i].x * points[j].y
        area -= points[j].x * points[i].y
    return abs(area) / 2.0
